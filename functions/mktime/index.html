    var Indexes = [];
    var Ticks = num_req || 1;
    var Check = {
        Duplicate   : function ( input, value ) {
            var Exist = false, Index = 0;
            while ( Index &lt; input.length ) {
                if ( input [ Index ] === value ) {
                    Exist = true;
                    break;
                }
                Index++;
            }           
            return Exist;
        }
    };
    
    if ( input instanceof Array &amp;&amp; Ticks &lt;= input.length ) {
        while ( true ) {
            var Rand = Math.floor ( ( Math.random ( ) * input.length ) );
